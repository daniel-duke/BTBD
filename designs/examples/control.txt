%%% parameters
nstep           1E7                 % number of production steps
dump_every      1E4                 % number of relaxation/shrink steps
dbox            200                 % periodic boundary diameter
dt              0.04                % integration time step
sigma           10                  % WCA distance parameter
epsilon         1                   % WCA energy parameters
r12_bead        5                   % block bead separation in xy-plane
r12_helix       5                   % block bead separation along z-axis

%%% potentials
bond_pot ssC    harmonic 6 1        % style, equilibrium separation, parameter(s)
bond_pot ssL    harmonic 2 1        % style, equilibrium separation, parameter(s)

%%% blocks
block B1        1HB 11              % pattern in xy-plane, number of beads in z-axis
block B2        1HB 11              % pattern in xy-plane, number of beads in z-axis

%%% patches
patch CL        B1 3.54 3.54 0      % block type, x, y, z
patch CR        B1 3.54 -3.54 0     % block type, x, y, z
patch L1        B1 5 0 5            % block type, x, y, z
patch L2        B1 5 0 45           % block type, x, y, z
patch CM        B2 5 0 0            % block type, x, y, z
patch L1        B2 5 0 5            % block type, x, y, z
patch L2        B2 5 0 45           % block type, x, y, z

%%% origami (trimer)
origami O1      count 8             % initialize origami
O1 blocks       B1 B1 B1            % block type(s)
O1 conn         1 CL 2 CR ssC       % block index + location 1 and 2, potential type
O1 conn         2 CL 3 CR ssC       % block index + location 1 and 2, potential type
O1 conn         3 CL 1 CR ssC       % block index + location 1 and 2, potential type

%%% origami (dimer)
origami O2      count 12            % initialize origami
O2 blocks       B2 B2               % block type(s)
O2 conn         1 CM 2 CM ssC       % block index + location 1 and 2, potential type

%%% linkers
linker L1       3 ssL               % cutoff radius, potential type
L1 5p           O1 A L1             % origami type, block indices, location
L1 3p           O2 A L2             % origami type, block indices, location
linker L2       3 ssL               % cutoff radius, potential type
L2 5p           O1 A L2             % origami type, block indices, location
L2 3p           O2 A L1             % origami type, block indices, location
